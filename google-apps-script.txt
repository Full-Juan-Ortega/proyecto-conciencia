const SHEET_NAME = "Pedidos";

function doPost(e) {
  try {
    const params = e.parameter;
    const ss = SpreadsheetApp.getActiveSpreadsheet();
    let sheet = ss.getSheetByName(SHEET_NAME);
    
    if (!sheet) {
      sheet = ss.insertSheet(SHEET_NAME);
      sheet.appendRow(["Fecha", "Nombre", "Empresa", "Teléfono", "Localidad", "Dirección", "Notas", "Detalle", "Total"]);
    }
    
    sheet.appendRow([
      params.fecha || "",
      params.nombre || "",
      params.empresa || "",
      params.telefono || "",
      params.localidad || "",
      params.direccion || "",
      params.notas || "",
      params.detalle || "",
      params.total || ""
    ]);
    
    return ContentService.createTextOutput(JSON.stringify({success: true, message: "Pedido recibido"})).setMimeType(ContentService.MimeType.JSON);
  } catch (error) {
    return ContentService.createTextOutput(JSON.stringify({success: false, error: error.toString()})).setMimeType(ContentService.MimeType.JSON);
  }
}

function doGet(e) {
  return ContentService.createTextOutput("El script está activo. Usá POST para enviar pedidos.");
}
